3 Build Your Web API

Step 3.1: Create a Model

Create a new file Models/Product.cs:

C# 
namespace MyWebApi.Models;

public class Product
{
    public int Id { get; set; }
    public string Name { get; set; } = string.Empty;
    public decimal Price { get; set; }
    public string Description { get; set; } = string.Empty;
    public bool IsAvailable { get; set; } = true;
}

Step 3.2: Create a Controller

Create Controllers/ProductsController.cs:

C# 
using Microsoft.AspNetCore.Mvc;
using MyWebApi.Models;

namespace MyWebApi.Controllers;

[ApiController]
[Route("api/[controller]")]
public class ProductsController : ControllerBase
{
    private static readonly List<Product> _products = new()
    {
        new Product { Id = 1, Name = "Laptop", Price = 999.99m, Description = "High-performance laptop" },
        new Product { Id = 2, Name = "Mouse", Price = 29.99m, Description = "Wireless mouse" }
    };

    // GET: api/products
    [HttpGet]
    public ActionResult<IEnumerable<Product>> GetAll()
    {
        return Ok(_products);
    }

    // GET: api/products/1
    [HttpGet("{id}")]
    public ActionResult<Product> GetById(int id)
    {
        var product = _products.FirstOrDefault(p => p.Id == id);
        if (product == null)
            return NotFound();
        
        return Ok(product);
    }

    // POST: api/products
    [HttpPost]
    public ActionResult<Product> Create(Product product)
    {
        product.Id = _products.Max(p => p.Id) + 1;
        _products.Add(product);
        return CreatedAtAction(nameof(GetById), new { id = product.Id }, product);
    }

    // PUT: api/products/1
    [HttpPut("{id}")]
    public IActionResult Update(int id, Product product)
    {
        var existing = _products.FirstOrDefault(p => p.Id == id);
        if (existing == null)
            return NotFound();
        
        existing.Name = product.Name;
        existing.Price = product.Price;
        existing.Description = product.Description;
        existing.IsAvailable = product.IsAvailable;
        
        return NoContent();
    }

    // DELETE: api/products/1
    [HttpDelete("{id}")]
    public IActionResult Delete(int id)
    {
        var product = _products.FirstOrDefault(p => p.Id == id);
        if (product == null)
            return NotFound();
        
        _products.Remove(product);
        return NoContent();
    }
}

Step 3.3: Configure CORS (Important for Production)

Update Program.cs:

C# 
var builder = WebApplication.CreateBuilder(args);

// Add services to the container
builder.Services.AddControllers();
builder.Services.AddEndpointsApiExplorer();
builder.Services.AddSwaggerGen();

// Configure CORS
builder.Services.AddCors(options =>
{
    options.AddPolicy("AllowAll",
        policy =>
        {
            policy.AllowAnyOrigin()
                  .AllowAnyMethod()
                  .AllowAnyHeader();
        });
});

var app = builder.Build();

// Configure the HTTP request pipeline
if (app.Environment.IsDevelopment())
{
    app.UseSwagger();
    app.UseSwaggerUI();
}

app.UseCors("AllowAll");
app.UseAuthorization();
app.MapControllers();

app.Run();

5 GitHub Actions CI/CD Pipeline

Step 5.1: Create Workflow Directory

Terminal 
# Create .github/workflows directory
mkdir -p .github/workflows

# Create workflow file
touch .github/workflows/deploy.yml

Step 5.2: Create CI/CD Workflow

Create .github/workflows/deploy.yml:

YAML 
name: Build and Deploy

on:
  push:
    branches: [ main ]
  pull_request:
    branches: [ main ]

jobs:
  build:
    runs-on: ubuntu-latest
    
    steps:
    - name: Checkout code
      uses: actions/checkout@v4
    
    - name: Setup .NET
      uses: actions/setup-dotnet@v4
      with:
        dotnet-version: '8.0.x'
    
    - name: Restore dependencies
      run: dotnet restore
    
    - name: Build
      run: dotnet build --configuration Release --no-restore
    
    - name: Test
      run: dotnet test --no-build --verbosity normal
    
    - name: Publish
      run: dotnet publish -c Release -o ./publish
    
    - name: Upload artifact
      uses: actions/upload-artifact@v4
      with:
        name: webapp
        path: ./publish
  
  deploy:
    needs: build
    runs-on: ubuntu-latest
    if: github.ref == 'refs/heads/main'
    
    steps:
    - name: Download artifact
      uses: actions/download-artifact@v4
      with:
        name: webapp
        path: ./publish
    
    - name: Deploy to Hostinger VPS
      uses: appleboy/scp-action@master
      with:
        host: ${{ secrets.VPS_HOST }}
        username: ${{ secrets.VPS_USERNAME }}
        key: ${{ secrets.VPS_SSH_KEY }}
        source: "./publish/*"
        target: "/var/www/mywebapi"
    
    - name: Restart Application
      uses: appleboy/ssh-action@master
      with:
        host: ${{ secrets.VPS_HOST }}
        username: ${{ secrets.VPS_USERNAME }}
        key: ${{ secrets.VPS_SSH_KEY }}
        script: |
          sudo systemctl restart mywebapi.service

Step 5.3: Understanding the Workflow

This workflow does the following:

  • Triggers: On push or pull request to main branch
  • Build Job: Restores, builds, tests, and publishes your app
  • Deploy Job: Uploads files to VPS and restarts the service
  • Security: Uses GitHub Secrets for sensitive data
‚ö†Ô∏è Important: Don't commit the workflow yet! We need to set up GitHub Secrets first (covered in the Hostinger Deploy tab).

Step 6.4: Configure NGINX Reverse Proxy

Create NGINX configuration:

Bash 
# Create NGINX site configuration
sudo nano /etc/nginx/sites-available/mywebapi

Add this configuration:

NGINX 
server {
    listen 80;
    listen [::]:80;
    server_name YOUR_DOMAIN_OR_IP;

    location / {
        proxy_pass http://localhost:5000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection keep-alive;
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}
Bash 
# Enable the site
sudo ln -s /etc/nginx/sites-available/mywebapi /etc/nginx/sites-enabled/

# Test NGINX configuration
sudo nginx -t

# Reload NGINX
sudo systemctl reload nginx

Step 6.5: Create Systemd Service

This keeps your app running in the background:

Bash 
# Create service file
sudo nano /etc/systemd/system/mywebapi.service

Add this configuration:

INI 
[Unit]
Description=My Web API Application
After=network.target

[Service]
Type=notify
WorkingDirectory=/var/www/mywebapi
ExecStart=/usr/bin/dotnet /var/www/mywebapi/MyWebApi.dll
Restart=always
RestartSec=10
KillSignal=SIGINT
SyslogIdentifier=mywebapi
User=root
Environment=ASPNETCORE_ENVIRONMENT=Production
Environment=ASPNETCORE_URLS=http://localhost:5000

[Install]
WantedBy=multi-user.target
Bash 
# Reload systemd
sudo systemctl daemon-reload

# Enable service to start on boot
sudo systemctl enable mywebapi.service

7 Troubleshooting & Common Issues

Issue 1: GitHub Actions Failing

Problem: Build or deploy job fails

Solutions:

  • Check Actions tab for detailed error logs
  • Verify all GitHub Secrets are correctly set
  • Ensure .NET version in workflow matches your project
  • Verify SSH key has no passphrase

Issue 2: Application Not Starting on VPS

Problem: Service fails to start after deployment
Bash 
# Check detailed error logs
sudo journalctl -u mywebapi.service -n 50 --no-pager

# Verify file permissions
ls -la /var/www/mywebapi

# Test manual run
cd /var/www/mywebapi
dotnet MyWebApi.dll

Issue 3: 502 Bad Gateway Error

Problem: NGINX shows 502 error

Causes & Solutions:

  • Application not running: Check service status with sudo systemctl status mywebapi
  • Wrong port: Verify app listens on port 5000 (check ASPNETCORE_URLS)
  • Firewall: Ensure port 5000 is open for localhost

Issue 4: CORS Errors

Problem: Browser blocks API requests

Verify CORS is configured in Program.cs (see Step 3.3 in Build API tab)

Issue 5: Port Already in Use

Bash 
# Find process using port 5000
sudo lsof -i :5000

# Kill the process (replace PID)
sudo kill -9 PID

# Or change port in systemd service file
sudo nano /etc/systemd/system/mywebapi.service
# Change ASPNETCORE_URLS to use different port

Issue 6: SSL Certificate Issues

Problem: HTTPS not working or certificate errors
Bash 
# Test certificate renewal
sudo certbot renew --dry-run

# Check certificate status
sudo certbot certificates

# Force certificate renewal
sudo certbot renew --force-renewal

Issue 7: Database Connection Failures

Solutions:

  • Store connection strings in environment variables
  • Never commit connection strings to GitHub
  • Use appsettings.Production.json for production settings
  • Add connection string to systemd service Environment variables
